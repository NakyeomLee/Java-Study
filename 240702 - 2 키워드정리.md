# Object class
- java에서 적는 모든 class들은 Object class를 부모 클래스로 둠
- 상속받는다고 extends를 이용하거나 다른 어떤 방법으로도 명시적으로 표시하지 않아도 Object class는 모든 클래스의 부모 클래스

```
class Bird {
   // 자식 클래스만 접근 가능한 protected
   protected int weight;

   // 생성자
   public Bird(int weight) {
      super();
      this.weight = weight;
   }

   // getter setter
   public int getWeight() {
      return weight;
   }

   public void setWeight(int weight) {
      this.weight = weight;
   }
   
   // 난다 메소드
   public void fly() {
      System.out.println();
   }
   
   // 울음소리 메소드
   public void Sound() {
      System.out.println();
   }

   // equals 재정의 
   @Override
   // 파라미터에 부모 클래스가 들어있으면 모든 정보를 받을 수 있음
   public boolean equals(Object obj) {
      
      if (obj == null) {
         return false;
      }
      
      if (!(obj instanceof Bird)) {
         return false;
      }
      
      Bird down = (Bird) obj;
      
      return this.weight == down.weight;
   }

   // toStirng 재정의(TestObject 참고)
   @Override
   public String toString() {
      return "새의 몸무게 : " + weight;
   }
}

// main절
public class TestBird {
   public static void main(String[] args) {
      
      Eagle e = new Eagle(20);
      Duck d = new Duck(5);
      
      System.out.println(e.weight);
      System.out.println(d.weight);
      
      e.fly();
      e.Sound();
      
      d.fly();
      d.Sound();
   }
}
```

```
public class TestObject {
   public static void main(String[] args) {
   
      // 변수 선언 (Object : 타입, book : 이름)
      // Object는 book, str, duck의 부모클래스
      // java에서 적는 모든 class들은 Object class를 부모 클래스로 두고 있기 때문
      Object book = new Book("책", 200, "작가이름");
      Object str = new String("문자열");
      Object duck = new Duck(3);
      
      // Book class와 Duck class에 toString 정의를 해두지 않아서 java자체에서 정의돼있는 toString으로 출력됨
      // 그래서 재정의 필요
      // Bird class(Duck의 부모 클래스)에 toString Override 해놓음
      System.out.println(book.toString()); 
      System.out.println(str.toString());
      System.out.println(duck.toString());
   }
}
```

```
public class TestObject2 {
   public static void main(String[] args) {
      Object i = new Integer(100);
      Object j = new Integer(100);
      
      String str1 = new String("Hello");
      String str2 = new String("Hello");
      
      System.out.println(i == j); // false
      
      System.out.println(str1 == str2); // false
      
      // equals => String의 동등함을 비교
      System.out.println(str1.equals(str2)); // true
      System.out.println(i.equals(j)); // true
   }
}
```

```
public class TestObject3 {
   public static void main(String[] args) {
      Bird b1 = new Bird(3);
      Bird b2 = new Bird(3);
      
      System.out.println(b1 == b2); // false
      
      // equals => String의 동등함을 비교
      // TestBird에 equals 재정의 해놓음
      // 재정의 해놓지않으면 false 출력됨
      // 객체의 동등함에 대한건 java도 java 만든사람도 아무도 모름
      // 오직 class를 직접 작성한 사람만 알고, 재정의도 할 수 있음
      System.out.println(b1.equals(b2)); // true
      
      Bird b3 = new Bird(5);
      
      System.out.println(b1.equals(b3)); // false
      
      // TestBird에 null이면 false 출력되게 equals Override에 조건 만들어놓음
      System.out.println(b1.equals(null)); // false
      
      // false 출력되게 TestBird의 equals Override에 조건 만들어놓음
      System.out.println(b1.equals(new String("문자열"))); // false
      
      System.out.println("프로그램 정상 종료");
   }
}
```

```
class Shape {
   public void printArea() {
      // 넓이 구하는 공식도 제각각이고 
      // 삼각형, 직사각형이 가질 정보도 달라서
      System.out.println("도형이라는 도형은 넓이를 구할 수가 없네요");
   }
}

// 삼각형
// 높이 : int
// 밑변 : int
// 넓이 출력 기능 : void
class Triangle extends Shape {
   int length;
   int height;
   
   // 메소드 재정의
   // (부모 클래스에 정의해놓은 넓이 메소드를 자식 클래스에서 재정의)
   @Override
   public void printArea() {
      System.out.println("삼각형의 넓이는 " + (length * height / 2));
   }
}

// 직사각형
// 가로 : int
// 세로 : int
// 넓이 출력 기능 : void
class Rectangle extends Shape {
   int width;
   int height;
   
   // 메소드 재정의
   // (부모 클래스에 정의해놓은 넓이 메소드를 자식 클래스에서 재정의)
   @Override
   public void printArea() {
      System.out.println("직사각형의 넓이는 " + (width * height));
   }
}

public class TestShape {
   public static void main(String[] args) {
      Triangle triangle = new Triangle();
      triangle.length = 20;
      triangle.height = 20;
      
      Rectangle rectangle = new Rectangle();
      rectangle.width = 5;
      rectangle.height = 10;
      
      triangle.printArea();
      rectangle.printArea();
      
      // 다형성
      // 객체는 다양한 형태를 가질 수 있음
      // 자식 클래스가 부모클래스를 참조해서 사용 가능
      // 부모의 방향으로 형변환 되는것 -> upcasting
      // 다시 자식의 형태를 가지게되는것(되돌아가는것) -> downcasting
      // 삼각형이 도형을 참조
      // 직사각형이 도형을 참조
      Shape shape1 = triangle;
      Shape shape2 = rectangle;
      
      // 동작에 대한 메소드를 호출하면 각 자식 클래스에서 재정의된 메소드 호출됨
      shape1.printArea(); // triangle 넓이
      shape2.printArea(); // rectangle 넓이
   }
}
```

```
class Tourist {
   // 자식클래스는 접근 가능 protected
   protected String name;

   // 생성자
   public Tourist(String name) {
      super();
      this.name = name;
   }

   // 인사 메소드
   public void sayHello() {
      System.out.println("-- 관광객이 인사를 합니다. --");
   }
}

// 한국말로 인사를 할 수 있는 관광객
// 이름 : String
// 인사하기 : void
class KoreanTourist extends Tourist {

   // 부모 클래스를 이용한 생성자 생성
   // (alt+shift+r 이용해서 생성 가능)
   public KoreanTourist(String name) {
      super(name);
   }

   // 메소드 재정의 (alt+shift+r 이용해서 생성 가능)
   @Override
   public void sayHello() {
      System.out.println("안녕하세요. 제 이름은 " + name + "입니다.");
   }
}

//영어로 인사를 할 수 있는 관광객
//이름 : String
//인사하기 : void
class EnglishTourist extends Tourist {

   // 부모 클래스를 이용한 생성자 생성
   // (alt+shift+r 이용해서 생성 가능)
   public EnglishTourist(String name) {
      super(name);
   }

   // 메소드 재정의 (alt+shift+r 이용해서 생성 가능)
   @Override
   public void sayHello() {
      System.out.println("Hello. My name is " + name + ".");
   }
}

public class TestTourist {
   public static void main(String[] args) {
      Tourist[] tourists = new Tourist[] {
            new KoreanTourist("길동")
            , new EnglishTourist("Tom")
            , new KoreanTourist("둘리")
            , new EnglishTourist("Brad")
      };
      
      for (int i = 0; i < tourists.length; i++) {
         tourists[i].sayHello();
      }
   }
}
출력값) 안녕하세요. 제 이름은 길동입니다.
       Hello. My name is Tom.
       안녕하세요. 제 이름은 둘리입니다.
       Hello. My name is Brad.
```
